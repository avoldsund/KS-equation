\center{The Kuramoto-Sivashinsky equation}

\begin{align*}
u_t + u_{xx} + u_{xxxx} + uu_x &= 0 \\
u_t + u_{xx} + u_{xxxx} + \frac{1}{2}(u^2)_x &= 0
\end{align*}
